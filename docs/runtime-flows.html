3. execution/trigger.ts — wireTrigger()

Scripts/execution/trigger.ts

sequenceDiagram
    participant Caller
    participant trig as trigger.ts
    participant comp as component.ts
    participant walk as walk.ts
    participant exec as execute.ts
    participant DOM

    Caller->>trig: wireTrigger(trigger, reaction, components, signal)
    trig->>trig: opts = signal ? { signal } : undefined

    alt trigger.kind === "dom-ready"
        alt document.readyState === "complete" OR "interactive"
            trig->>exec: executeReaction(reaction, { components })
            Note over trig: Execute immediately, DOM is ready
        else readyState === "loading"
            trig->>DOM: document.addEventListener("DOMContentLoaded", handler, opts)
            Note over DOM: When DOMContentLoaded fires:
            DOM->>exec: executeReaction(reaction, { components })
        end

    else trigger.kind === "custom-event"
        trig->>trig: log.debug("custom-event: listening", { event })
        trig->>DOM: document.addEventListener(trigger.event, handler, opts)
        Note over DOM: When custom event fires:
        DOM->>DOM: detail = (e as CustomEvent).detail
        DOM->>exec: executeReaction(reaction, { evt: detail ?? {}, components })

    else trigger.kind === "component-event"
        trig->>DOM: document.getElementById(trigger.componentId)
        alt element NOT found
            Note right of trig: THROW Error("[alis] element not found: {componentId}")
        end
        DOM-->>trig: el

        trig->>comp: resolveRoot(el, trigger.vendor)
        comp-->>trig: root (EventTarget)

        trig->>trig: log.debug("component-event", { componentId, jsEvent, vendor })
        trig->>DOM: root.addEventListener(trigger.jsEvent, handler, opts)

        Note over DOM: When component event fires:
        alt trigger.vendor === "native"
            alt readExpr exists
                trig->>walk: walk(el, readExpr)
                walk-->>trig: value
                Note over trig: detail = { [readExpr]: value, event: e }
            else no readExpr
                Note over trig: detail = { event: e }
            end
        else vendor === "fusion"
            Note over trig: detail = e ?? {} (use event as-is)
        end
        DOM->>exec: executeReaction(reaction, { evt: detail, components })

    else default (unknown trigger kind)
        Note right of trig: THROW assertNever(trigger, "trigger kind")
    end

4. execution/execute.ts — executeReaction() + dispatchAsync()

Scripts/execution/execute.ts

4a. needsAsync() — Recursive ConfirmGuard Detection

sequenceDiagram
    participant exec as execute.ts
    participant cond as conditions.ts

    Note over exec: needsAsync(reaction)

    alt reaction.kind === "conditional"
        loop for each branch
            alt branch.guard != null AND isConfirmGuard(branch.guard)
                exec-->>exec: return true
            else check nested reaction
                exec->>exec: needsAsync(branch.reaction)
                alt nested returns true
                    exec-->>exec: return true
                end
            end
        end
        exec-->>exec: return false (no confirm found)
    else any other reaction kind
        exec-->>exec: return false
    end

4b. executeReaction() — Sync Path

sequenceDiagram
    participant Caller
    participant exec as execute.ts
    participant cmd as commands.ts
    participant cond as conditions.ts
    participant pipe as pipeline.ts

    Caller->>exec: executeReaction(reaction, ctx)

    alt needsAsync(reaction) returns true
        exec->>exec: dispatchAsync(reaction, ctx)
        exec-->>Caller: return (async path takes over)
    end

    Note over exec: Sync path — switch(reaction.kind)

    alt kind === "sequential"
        exec->>exec: log.debug("sequential", { commands.length })
        loop for each command
            exec->>cmd: executeCommand(cmd, ctx)
        end

    else kind === "conditional"
        exec->>exec: log.debug("conditional", { commands, branches })
        opt reaction.commands exists (preamble)
            loop for each preamble command
                exec->>cmd: executeCommand(cmd, ctx)
            end
        end
        loop for each branch
            alt branch.guard == null (else branch)
                exec->>exec: log.trace("branch-taken", { guard: "else" })
                exec->>exec: executeReaction(branch.reaction, ctx)
                exec-->>Caller: return (first match wins)
            else evaluateGuard(branch.guard, ctx) === true
                exec->>cond: evaluateGuard(branch.guard, ctx)
                cond-->>exec: true
                exec->>exec: log.trace("branch-taken", { guard })
                exec->>exec: executeReaction(branch.reaction, ctx)
                exec-->>Caller: return (first match wins)
            else evaluateGuard returns false
                exec->>cond: evaluateGuard(branch.guard, ctx)
                cond-->>exec: false
                Note over exec: continue to next branch
            end
        end
        exec->>exec: log.trace("no-branch-taken")

    else kind === "http"
        exec->>exec: log.debug("http", { url })
        exec->>pipe: executeHttpReaction(reaction, ctx)

    else kind === "parallel-http"
        exec->>exec: log.debug("parallel-http", { count })
        exec->>pipe: executeParallelHttpReaction(reaction, ctx)

    else default
        Note right of exec: THROW assertNever(reaction, "reaction kind")
    end

4c. dispatchAsync() — Async Path (ConfirmGuard Present)

sequenceDiagram
    participant exec as execute.ts
    participant cmd as commands.ts
    participant cond as conditions.ts
    participant pipe as pipeline.ts

    Note over exec: dispatchAsync(reaction, ctx) - async

5. execution/commands.ts — executeCommand()

Scripts/execution/commands.ts

sequenceDiagram
    participant Caller
    participant cmds as commands.ts
    participant cond as conditions.ts
    participant elem as element.ts
    participant val as validation/index.ts
    participant inject as inject.ts
    participant DOM

    Caller->>cmds: executeCommand(cmd, ctx)

    alt cmd.when guard exists
        cmds->>cond: isConfirmGuard(cmd.when)
        alt is ConfirmGuard
            cmds->>cmds: log.warn("ConfirmGuard on per-action When not supported")
            cmds-->>Caller: return (skip command)
        end

        cmds->>cond: evaluateGuard(cmd.when, ctx)
        alt returns false
            cmds->>cmds: log.trace("per-action-when-skipped", { kind })
            cmds-->>Caller: return (skip command)
        end
        Note over cmds: Guard passed, continue to command execution
    end

    Note over cmds: switch(cmd.kind)

    alt kind === "dispatch"
        cmds->>cmds: log.trace("dispatch", { event, payload })
        cmds->>DOM: document.dispatchEvent(new CustomEvent(cmd.event, { detail: payload ?? {} }))

    else kind === "mutate-element"
        cmds->>cmds: log.trace("mutate-element", { target, mutation.kind })
        cmds->>elem: mutateElement(cmd, ctx)

    else kind === "validation-errors"
        alt ctx.responseBody is falsy
            Note over cmds: break (no response body to display)
        else responseBody exists
            alt ctx.validationDesc is falsy
                Note right of cmds: THROW Error('[alis] ValidationErrors("{formId}") requires a validation descriptor.')
            else validationDesc exists
                cmds->>val: showServerErrors(ctx.validationDesc, ctx.responseBody)
            end
        end

    else kind === "into"
        cmds->>DOM: document.getElementById(cmd.target)
        DOM-->>cmds: container (or null)
        alt container exists AND ctx.responseBody != null
            cmds->>inject: injectHtml(container, String(ctx.responseBody))
        else container null OR no responseBody
            Note over cmds: No-op (silent skip)
        end

    else default
        Note right of cmds: THROW assertNever(cmd, "command kind")
    end

executeCommands(commands, ctx) simply loops over commands and calls executeCommand for each.

7. conditions/conditions.ts — evaluateGuard() + evaluateValueGuard()

Scripts/conditions/conditions.ts

7a. isConfirmGuard() — Recursive Check

sequenceDiagram
    participant Caller
    participant cond as conditions.ts

    Caller->>cond: isConfirmGuard(guard)

    alt guard.kind === "confirm"
        cond-->>Caller: return true
    else guard.kind === "not"
        cond->>cond: isConfirmGuard(guard.inner)
        cond-->>Caller: return result
    else guard.kind === "all" or "any"
        cond->>cond: guard.guards.some(isConfirmGuard)
        cond-->>Caller: return result
    else guard.kind === "value"
        cond-->>Caller: return false
    end

7b. evaluateGuard() — Sync

sequenceDiagram
    participant Caller
    participant cond as conditions.ts

    Caller->>cond: evaluateGuard(guard, ctx)

    alt guard.kind === "value"
        cond->>cond: evaluateValueGuard(guard, ctx)
        cond-->>Caller: return boolean

    else guard.kind === "all"
        cond->>cond: guard.guards.every(g => evaluateGuard(g, ctx))
        cond-->>Caller: return boolean

    else guard.kind === "any"
        cond->>cond: guard.guards.some(g => evaluateGuard(g, ctx))
        cond-->>Caller: return boolean

    else guard.kind === "not"
        cond->>cond: !evaluateGuard(guard.inner, ctx)
        cond-->>Caller: return boolean

    else guard.kind === "confirm"
        cond->>cond: log.warn("ConfirmGuard in sync context - denying")
        cond-->>Caller: return false

    else default
        Note right of cond: THROW assertNever(guard, "guard kind")
    end

7c. evaluateGuardAsync() — Async

sequenceDiagram
    participant Caller
    participant cond as conditions.ts
    participant win as window.alis

    Caller->>cond: await evaluateGuardAsync(guard, ctx)

    alt guard.kind === "value"
        cond->>cond: evaluateValueGuard(guard, ctx)
        cond-->>Caller: return boolean

    else guard.kind === "all"
        loop for each guard in guards (sequential)
            cond->>cond: await evaluateGuardAsync(g, ctx)
            alt returns false
                cond-->>Caller: return false (short-circuit)
            end
        end
        cond-->>Caller: return true

    else guard.kind === "any"
        loop for each guard in guards (sequential)
            cond->>cond: await evaluateGuardAsync(g, ctx)
            alt returns true
                cond-->>Caller: return true (short-circuit)
            end
        end
        cond-->>Caller: return false

    else guard.kind === "not"
        cond->>cond: await evaluateGuardAsync(guard.inner, ctx)
        cond-->>Caller: return negated result

    else guard.kind === "confirm"

        alt alis.confirm exists
            win-->>cond: Promise of boolean (user clicks OK/Cancel)

    else default

7d. evaluateValueGuard() — All Operators

sequenceDiagram
    participant cond as conditions.ts
    participant res as resolver.ts

    Note over cond: evaluateValueGuard(guard, ctx)

    alt op === "is-null" or "not-null"
        cond->>res: resolveSource(guard.source, ctx)
        res-->>cond: raw (NO coercion)
        cond->>cond: log.trace("eval-presence")
        alt op === "is-null"
            cond-->>cond: return raw == null
        else op === "not-null"
            cond-->>cond: return raw != null
        end

    else op === "is-empty" or "not-empty"
        cond->>res: resolveSource(guard.source, ctx)
        res-->>cond: raw (NO coercion)
        cond->>cond: log.trace("eval-presence")
        Note over cond: isEmpty = raw==="" || raw===null || raw===undefined || (isArray && length===0)
        alt op === "is-empty"
            cond-->>cond: return isEmpty
        else op === "not-empty"
            cond-->>cond: return !isEmpty
        end

    else all other operators
        cond->>res: resolveSourceAs(guard.source, guard.coerceAs, ctx)
        res-->>cond: resolved (coerced left side)

        Note over cond: Operand coercion: opCoerceAs = elementCoerceAs ?? coerceAs

        alt guard.rightSource present
            cond->>res: resolveSourceAs(guard.rightSource, opCoerceAs, ctx)
            res-->>cond: operand (coerced right side from source)
        else literal operand
            alt operand is Array
                Note over cond: operand = rawOp.map(v => coerce(v, opCoerceAs))
            else operand is scalar
                cond->>res: coerce(rawOp, opCoerceAs)
                res-->>cond: operand
            end
        end

        opt elementCoerceAs present AND resolved is Array
            Note over cond: items = resolved.map(item => coerce(item, elementCoerceAs))
        end

        cond->>cond: log.trace("eval", { source, op, resolved, operand })

        alt op === "eq"
            cond-->>cond: return resolved === operand
        else op === "neq"
            cond-->>cond: return resolved !== operand
        else op === "gt"
            cond-->>cond: return resolved > operand
        else op === "gte"
            cond-->>cond: return resolved >= operand
        else op === "lt"
            cond-->>cond: return resolved < operand
        else op === "lte"
            cond-->>cond: return resolved <= operand
        else op === "truthy"
            cond-->>cond: return !!resolved
        else op === "falsy"
            cond-->>cond: return !resolved
        else op === "in"
            cond-->>cond: return isArray(operand) && operand.includes(resolved)
        else op === "not-in"
            cond-->>cond: return !isArray(operand) || !operand.includes(resolved)
        else op === "between"
            cond-->>cond: return isArray(operand) && resolved >= [0] && resolved <= [1]
        else op === "array-contains"
            cond-->>cond: return items != null && items.includes(operand)
        else op === "contains"
            cond-->>cond: return String(resolved).includes(String(operand))
        else op === "starts-with"
            cond-->>cond: return String(resolved).startsWith(String(operand))
        else op === "ends-with"
            cond-->>cond: return String(resolved).endsWith(String(operand))
        else op === "matches"
            cond->>cond: try new RegExp(operand).test(resolved)
            alt regex succeeds
                cond-->>cond: return test result
            else regex throws (invalid pattern)
                cond->>cond: log.warn("invalid guard regex")
                cond-->>cond: return false
            end
        else op === "min-length"
            cond-->>cond: return String(resolved).length >= Number(operand)
        else default (unknown op)
            Note right of cond: THROW Error("Unknown guard operator: {op}")
        end
    end

sequenceDiagram

        comp-->>res: value

8c. coerce() — All Coercion Types

sequenceDiagram
    participant Caller
    participant res as resolver.ts

    Caller->>res: coerce(value, coerceAs)

    alt coerceAs === "string"
        Note over res: String(value ?? "")
        Note over res: null/undefined -> ""
        res-->>Caller: return string

    else coerceAs === "number"
        Note over res: n = Number(value)
        alt Number.isNaN(n)
            res-->>Caller: return 0
        else valid number
            res-->>Caller: return n
        end

    else coerceAs === "boolean"
        alt typeof value === "string"
            Note over res: return value !== "" && value !== "false" && value !== "0"
            res-->>Caller: return boolean
        else non-string
            Note over res: return Boolean(value)
            res-->>Caller: return boolean
        end

    else coerceAs === "date"
        alt value == null
            res-->>Caller: return NaN
        else value exists
            Note over res: d = new Date(value as string)
            res-->>Caller: return d.getTime()
        end

    else coerceAs === "array"
        alt Array.isArray(value)
            res-->>Caller: return value (pass-through)
        else value == null or value === ""
            res-->>Caller: return [] (empty array)
        else scalar value
            res-->>Caller: return [value] (wrap in array)
        end

    else coerceAs === "raw"
        res-->>Caller: return value (no coercion)
    end

11a. execRequest()

sequenceDiagram
    participant Caller
    participant http as http.ts
    participant cmds as commands.ts
    participant gather as gather.ts
    participant exec as execute.ts
    participant Network

    Caller->>http: await execRequest(req, ctx)

    Note over http: Step 1: WhileLoading
    opt req.whileLoading exists
        http->>cmds: executeCommands(req.whileLoading, ctx)
    end

    Note over http: Step 2: Gather
    http->>gather: resolveGather(req.gather ?? [], req.verb, ctx.components ?? {}, req.contentType)
    gather-->>http: gatherResult { urlParams, body }

    Note over http: Step 3: Build fetch options
    http->>http: url = req.url
    http->>http: init = { method: req.verb }

    alt gatherResult.urlParams.length > 0
        alt url includes "?"
            Note over http: url = url + "&" + params.join("&")
        else no existing query string
            Note over http: url = url + "?" + params.join("&")
        end
    end

    alt req.verb !== "GET"
        alt body instanceof FormData
            Note over http: init.body = formData
(browser auto-sets Content-Type
with multipart boundary)
        else body has keys (JSON object)
            Note over http: init.headers = { "Content-Type": "application/json" }
init.body = JSON.stringify(body)
        else body is empty object
            Note over http: No body attached
        end
    end

    http->>http: log.debug("fetch", { verb, url })

    Note over http: Step 4: Execute fetch
    http->>Network: fetch(url, init)

    alt fetch succeeds (no network error)
        Network-->>http: response

        Note over http: Step 5: Read response body
        http->>http: readResponseBody(response)

        alt Content-Type includes "application/json"
            http->>http: response.json()
        else Content-Type includes "text/"
            http->>http: response.text()
        else Content-Type includes "html"
            http->>http: response.text()
        else no matching Content-Type
            Note over http: body = null
        end

        alt response.ok (status 2xx)
            alt body != null
                Note over http: successCtx = { ...ctx, responseBody: body }
            else body is null
                Note over http: successCtx = ctx (no responseBody)
            end
            http->>http: routeHandlers(req.onSuccess, response.status, successCtx)
        else response NOT ok (4xx, 5xx)
            Note over http: errorCtx = { ...ctx, responseBody: body ?? undefined, validationDesc: req.validation }
            http->>http: routeHandlers(req.onError, response.status, errorCtx)
        end

        Note over http: Step 6: Chained request
        alt req.chained exists AND response.ok
            http->>http: await execRequest(req.chained, ctx)
        end

    else fetch throws (network error)
        http->>http: log.error("network error", { url, error })
        http->>http: routeHandlers(req.onError, 0, ctx)
        Note over http: status 0 = network failure
    end
